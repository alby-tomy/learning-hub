Part A — User Role System

A single User model (extending AbstractBaseUser) powers every person in the system. A role field on the user decides what they can see and do. Each role has a separate profile model that stores role-specific details — keeping the auth table clean and generic.

from django.contrib.auth.models import AbstractBaseUser, BaseUserManager, PermissionsMixin
from django.db import models

class SchoolUserManager(BaseUserManager):
    def create_user(self, email, password=None, **extra):
        if not email: raise ValueError('Email is required')
        user = self.model(email=self.normalize_email(email), **extra)
        user.set_password(password)
        user.save(using=self._db)
        return user

    def create_superuser(self, email, password=None, **extra):
        extra.setdefault('is_staff', True)
        extra.setdefault('is_superuser', True)
        extra.setdefault('role', 'admin')
        return self.create_user(email, password, **extra)


class User(AbstractBaseUser, PermissionsMixin):
    # ── Role choices — covers every actor in the system ────────────────
    ROLES = [
        ('admin',       'Administrator'),
        ('teacher',     'Teacher'),
        ('student',     'Student'),
        ('non_teaching','Non-Teaching Staff'),
        ('parent',      'Parent'),
        ('guest',       'Guest'),
    ]

    email     = models.EmailField(unique=True)          # login identifier
    full_name = models.CharField(max_length=150)
    role      = models.CharField(max_length=15, choices=ROLES, default='student')
    phone     = models.CharField(max_length=15, blank=True)
    avatar    = models.ImageField(upload_to='avatars/', null=True, blank=True)
    is_active = models.BooleanField(default=True)
    is_staff  = models.BooleanField(default=False)     # True → Django admin access
    date_joined = models.DateTimeField(auto_now_add=True)

    USERNAME_FIELD  = 'email'
    REQUIRED_FIELDS = ['full_name', 'role']
    objects = SchoolUserManager()

    # ── Convenience properties to check role without importing constants ──
    @property
    def is_admin(self):         return self.role == 'admin'
    @property
    def is_teacher(self):       return self.role == 'teacher'
    @property
    def is_student(self):       return self.role == 'student'
    @property
    def is_non_teaching(self):  return self.role == 'non_teaching'
    @property
    def is_parent(self):        return self.role == 'parent'

    def __str__(self): return f"{self.full_name} [{self.role}]"

Part B — Every Model Explained

① Administrator

Administrators are just User rows with role='admin' and is_staff=True. They get full access to the Django admin panel (/admin/) and all API endpoints. No separate profile model is needed — the User table is enough.

FieldPurpose
User.role = 'admin'Grants elevated permissions in every permission class
User.is_staff = TrueUnlocks the Django /admin/ interface
User.is_superuser = TrueBypasses all permission checks (superadmin only)

② Teacher

A teacher has a login account (User with role='teacher') linked one-to-one to a Teacher profile that stores professional details. A teacher can teach multiple subjects and be assigned to one or more classrooms via the timetable.

teachers/models.py — TeacherPython
class Teacher(models.Model):
    # OneToOne → User means ONE teacher account per person
    user          = models.OneToOneField(
        settings.AUTH_USER_MODEL,
        on_delete=models.CASCADE,
        related_name='teacher_profile',
        limit_choices_to={'role': 'teacher'},   # admin form only shows teacher-role users
    )
    employee_id   = models.CharField(max_length=20, unique=True)   # 'TCH-2024-001'
    qualification = models.CharField(max_length=200)                 # 'B.Ed, M.Sc Physics'
    joining_date  = models.DateField()
    salary        = models.DecimalField(max_digits=10, decimal_places=2)
    is_active     = models.BooleanField(default=True)

    # A teacher can teach MANY subjects (M2M)
    subjects      = models.ManyToManyField('Subject', blank=True, related_name='teachers')

    class Meta:
        ordering = ['employee_id']

    def __str__(self): return f"{self.user.full_name} ({self.employee_id})"

# Access patterns:
teacher = Teacher.objects.select_related('user').get(employee_id='TCH-2024-001')
teacher.user.full_name         # 'Mr. John Smith'
teacher.user.email             # 'john@school.com'
teacher.subjects.all()         # all subjects this teacher teaches
teacher.classrooms.all()       # classrooms where this teacher is class_teacher (reverse FK)

③ Class Teacher

Class Teacher is not a separate model. It is a role a Teacher plays — being the primary in-charge of a specific Classroom. Django models this as a ForeignKey on Classroom pointing to Teacher. One teacher can be the class teacher of at most one class per academic year (enforced via unique_together).

classes/models.py — Classroom with class_teacher FKPython
class Classroom(models.Model):
    name          = models.CharField(max_length=20, unique=True)   # '10-A'
    grade         = models.PositiveSmallIntegerField()                 # 10
    section       = models.CharField(max_length=5)                   # 'A'
    academic_year = models.CharField(max_length=9, default='2024-2025')
    max_capacity  = models.PositiveSmallIntegerField(default=40)

    # ★ Class Teacher — one teacher in-charge per classroom
    class_teacher = models.ForeignKey(
        'teachers.Teacher',
        on_delete=models.SET_NULL,
        null=True, blank=True,
        related_name='class_in_charge',    # teacher.class_in_charge.all() → Classrooms they manage
    )

    # Subjects taught in this classroom
    subjects      = models.ManyToManyField('teachers.Subject', blank=True, related_name='classrooms')

    class Meta:
        unique_together = [('grade', 'section', 'academic_year')]
        ordering        = ['grade', 'section']

    def __str__(self): return self.name


# Access patterns:
room = Classroom.objects.select_related('class_teacher__user').get(name='10-A')
room.class_teacher.user.full_name    # 'Mrs. Alice'
room.students.all()                  # all students (reverse FK from Student)
room.subjects.all()                  # subjects taught in this class

# All classrooms a teacher is class teacher of:
teacher = Teacher.objects.get(employee_id='TCH-001')

# Check if a teacher is a class teacher of any class:

④ Student

A student has a login account (User with role='student') linked to a Student profile. The profile stores school-specific data: roll number, classroom assignment, GPA. The student belongs to one Classroom (ForeignKey) and can have one or more Parent records linked to them.

students/models.py — StudentPython
class Student(models.Model):
    # OneToOne → User (student login account)
    user      = models.OneToOneField(
        settings.AUTH_USER_MODEL,
        on_delete=models.CASCADE,
        related_name='student_profile',
        limit_choices_to={'role': 'student'},
    )
    roll_no   = models.SlugField(unique=True)        # '2024-10A-001'
    dob       = models.DateField()
    gender    = models.CharField(max_length=1, choices=[('M','Male'),('F','Female'),('O','Other')])
    photo     = models.ImageField(upload_to='students/photos/%Y/%m/', null=True, blank=True)
    address   = models.TextField(blank=True)
    is_active = models.BooleanField(default=True)
    gpa       = models.DecimalField(max_digits=4, decimal_places=2, default=0.00)
    admitted_on = models.DateField(auto_now_add=True)

    # FK → Classroom — a student belongs to ONE class
    classroom = models.ForeignKey(
        'classes.Classroom',
        on_delete=models.SET_NULL,
        null=True,
        related_name='students',
    )

    class Meta:
        ordering = ['roll_no']
        indexes  = [models.Index(fields=['roll_no'])]

    def __str__(self): return f"{self.roll_no} — {self.user.full_name}"

# Access patterns:
s = Student.objects.select_related('user', 'classroom').get(roll_no='2024-10A-001')

s.classroom.name          # '10-A'

⑦ Parent / Guardian

A Parent is linked to one or more students via ForeignKey. A parent can have a login account (User with role='parent') to log in and view their child's data — or they can be stored as contact-only without an account. The user field is optional (null=True).

Part D — Guest Access

A Guest has no account. In Django REST Framework, guest access means AllowAny permission on public endpoints. No model is needed — the role is expressed in the API permission class, not the database.

How guest vs authenticated roles translate to DRF permissionsPython
from rest_framework.permissions import BasePermission, IsAuthenticated, AllowAny

class IsAdminRole(BasePermission):
    def has_permission(self, request, view):
        return request.user.is_authenticated and request.user.role == 'admin'

class IsTeacherRole(BasePermission):
    def has_permission(self, request, view):
        return request.user.is_authenticated and request.user.role == 'teacher'

class IsStudentRole(BasePermission):
    def has_permission(self, request, view):
        return request.user.is_authenticated and request.user.role == 'student'

class IsParentRole(BasePermission):
    def has_permission(self, request, view):
        return request.user.is_authenticated and request.user.role == 'parent'

class IsNonTeachingRole(BasePermission):
    def has_permission(self, request, view):
        return request.user.is_authenticated and request.user.role == 'non_teaching'


# In a ViewSet — combine permissions per action
class StudentViewSet(viewsets.ModelViewSet):
    def get_permissions(self):
        if self.action in ['list', 'retrieve']:
            # admins, teachers, parents can view students
            return [(IsAdminRole | IsTeacherRole | IsParentRole)()]
        if self.action in ['create', 'update', 'destroy']:
            return [IsAdminRole()]      # only admins can modify students
        return [IsAuthenticated()]      # default: any logged-in user

# Public endpoint (guest access) — no auth needed
class SchoolInfoView(APIView):
    permission_classes = [AllowAny]   # guests can see school public info

    message = 'Only teachers or admins can perform this action.'
    def has_permission(self, request, view):
        if request.method in permissions.SAFE_METHODS:
            return request.user.is_authenticated
        return (request.user.is_authenticated and
                request.user.role in ['teacher', 'admin'] or
                request.user.is_staff)

    def has_object_permission(self, request, view, obj):
        if request.method in permissions.SAFE_METHODS: return True
        if request.user.is_staff or request.user.role == 'admin': return True
        # Teacher can only modify students in their classroom

from rest_framework.permissions import BasePermission, SAFE_METHODS

class IsSchoolAdmin(BasePermission):
    def has_permission(self, request, view):
        return bool(
            request.user and request.user.is_authenticated and
            (request.user.role == 'admin' or request.user.is_staff)
        )


class IsTeacherOrAdmin(BasePermission):
    def has_permission(self, request, view):
        if request.method in SAFE_METHODS: return request.user.is_authenticated
        return request.user.is_authenticated and (
            request.user.role in ('teacher', 'admin') or request.user.is_staff

            request.user.is_staff or request.user.role in ('teacher', 'admin')
        )

    def has_object_permission(self, request, view, obj):
        if request.method in SAFE_METHODS: return True
        if request.user.is_staff or request.user.role == 'admin': return True
        classroom = getattr(obj, 'classroom', None)
        return classroom and classroom.teacher == request.user

        if request.method not in SAFE_METHODS: return False
        if request.user.role != 'parent': return True
        guardian = getattr(obj, 'guardian', None)
        return guardian and guardian.user == request.user
